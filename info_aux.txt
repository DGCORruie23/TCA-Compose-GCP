from usuarios.models import Registro
from django.db.models import Count
from django.db.models.functions import TruncMonth


queryset = (
    Registro.objects.filter(estado=1)
    .annotate(mes=TruncMonth('fecha_finalizacion'))
    .values('mes', 'claveAcuerdo')
    .annotate(total=Count('idRegistro'))
    .order_by('mes')
)

queryset2 = (
    Registro.objects.filter(estado=2)
    .annotate(mes=TruncMonth('fecha_inicio'))
    .values('mes')
    .annotate(total=Count('idRegistro'))
    .order_by('mes')
)




upstream backendTCA{
    server web:8000;
}

server {

        listen 80 default_server;
        client_max_body_size 100M;

        location / {
            proxy_pass http://backendTCA;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header Host $host;
            proxy_redirect off;
        }

        location /static/ {
            alias /src/TCA/data/static/;
        }

        location /media/ {
            alias /src/TCA/data/media/;
        }

}

events {
    worker_connections 1024;
}

http {
    upstream backendTCA {
        server web:8000;
    }

    server {
        listen 80;
        client_max_body_size 100M;

        location ~ /.well-known/acme-challenge/ {
            root /var/www/certbot/;
        }

        return 301 https://$host$request_uri;
    }

    server {
        listen 443 ssl;
        http2 on;
        client_max_body_size 100M;

        ssl_certificate /etc/letsencrypt/live/acuerdos.dgcor.com/fullchain.pem;
        ssl_certificate_key /etc/letsencrypt/live/acuerdos.dgcor.com/privkey.pem;

        location / {
            proxy_pass http://backendTCA;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header Host $host;
            proxy_redirect off;
        }

        location /static/ {
            root /src/TCA/data/static/;

        location /media/ {
            root /src/TCA/data/media/;
        }

            root /var/www/certbot/;
        }
    }
}
